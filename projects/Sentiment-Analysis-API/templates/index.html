API Reference

POST /api/analyze VADER sentiment 
# Request
curl -X POST https://your-host/api/analyze \
  -H "Content-Type: application/json" \
  -d '{"text": "This product is incredible!"}'

# Response
{
  "label":    "POSITIVE",
  "score":     0.8316,
  "compound":  0.8316,
  "positive":  0.625,
  "neutral":   0.375,
  "negative":  0.0,
  "backend":  "nltk-vader",
  "input_length": 27
}
POST /api/analyze/hf DistilBERT sentiment 
# Request (same shape)
curl -X POST https://your-host/api/analyze/hf \
  -H "Content-Type: application/json" \
  -d '{"text": "Absolutely terrible experience."}'

# Response
{
  "label":    "NEGATIVE",
  "score":     0.9987,
  "compound": -0.9987,
  "backend":  "huggingface-distilbert",
  "input_length": 31
}
POST /api/analyze/batch Batch (up to 100 texts) 
# Request
curl -X POST https://your-host/api/analyze/batch \
  -H "Content-Type: application/json" \
  -d '{"texts": ["Love it!", "Hate it.", "Meh."]}'

# Response
{
  "count": 3,
  "results": [
    {"label": "POSITIVE", "score": 0.6696, ...},
    {"label": "NEGATIVE", "score": 0.5719, ...},
    {"label": "NEUTRAL",  "score": 0.0,    ...}
  ]
}
